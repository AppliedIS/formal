%extend_1 {
    color: $theme3-gray-light;
    font-size: 0.75em;
    top: -18px;
    transition: all 0.225s ease;
}

input {
    background: none;
    border-color: $theme3-blue-dark;
    border-radius: 2px;
    border-width: 0 0 2px 0;
    color: $clr-white;
    font-size: 1rem;
    padding: 18px 10px 10px 10px;
    width: 100%;

    &:focus {
        border-color: $theme3-gray-light;
        outline: 0;

        ~ label {
            @extend %extend_1;
        }

        ~ span {
            transition: all 0.075s ease;
            width: 100%;
        }
    }

    &:valid ~ label {
        @extend %extend_1;
    }
}
